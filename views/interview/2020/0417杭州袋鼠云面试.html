    # 求数组的最大值（四种方式）

    # 1.forEach遍历 把最大值存储起来

    本质还是for循环，无返回值，若想改变原数组，使用arr[i]=新值，即可改变

    let arr = [2,32,4,5,3,23];
let max = arr[0];
arr.forEach((item) => {
  max = item > max ? item : max;
})
console.log(max);

    # 2.使用sort()将数组排序

    a-b从小到大排序，b-a从大到小排序

    let arr = [2,32,4,5,3,23];
arr.sort((a , b) => {
  return b - a;

    # 3.使用Math.max(),借用apply

    Math.max()只能传入多个参数，不能直接传入数组 可以借用apply或者扩展运算符

    //使用apply
let arr = [2,32,4,5,3,23];
Math.max.apply(null, arr);
